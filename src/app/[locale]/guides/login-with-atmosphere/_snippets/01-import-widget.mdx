```typescript
import { AtmosphereLogin } from '@atproto/atmosphere-login'
import { Agent } from '@atproto/lex'

// Returns a full authenticated session
const session = await AtmosphereLogin.prompt({
  scope: 'atproto transition:generic',
})

console.log(session.did)    // 'did:plc:...'
console.log(session.handle) // 'alice.bsky.social'

// Use the session for API calls
const agent = new Agent(session)
await agent.post({ text: 'Hello from Atmosphere!' })
```
