import (
  "context"
  "github.com/bluesky-social/indigo/atproto/atclient"
  "github.com/bluesky-social/indigo/atproto/identity"
)

func main() {
  ctx := context.Background()

  client, _ := atclient.LoginWithPassword(ctx,
    identity.DefaultDirectory(),   // handles PDS lookup automatically
    "your-handle.bsky.social",     // identifier
    "your-app-password",           // password
    "",                            // authToken (optional 2FA token)
    nil,                           // refresh callback (optional)
  )
}
